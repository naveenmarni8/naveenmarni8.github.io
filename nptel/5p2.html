//Create a class B, which implements the interface GCD.
class B implements GCD
{
  public int findGCD(int n1,int n2)
  {
  if(n1<0||n2<0)
    return -1;
  else
  {
   if(n2==0)
     return n1;
    return  findGCD(n2,n1%n2);
  
  }
  }
}
